DATA EXFILTRATION

Export intel from Nautobot to CSV dropfiles

// EXTRACTION PROTOCOL

DataFlow needs network intel extracted as CSV. You'll query the database and generate structured export files.

PHASE 1: QUERY DATABASE

Use Django QuerySets to retrieve data:

QuerySet Examples 
from nautobot.dcim.models import Device

# Get all devices
devices = Device.objects.all()

# Filter by criteria
dc_devices = Device.objects.filter(location__name="TechNova DC")

# Multiple filters
active_switches = Device.objects.filter(
    role__name="Access Switch",
    status__name="Active"
)

# Optimize with select_related (eager load FKs)
devices = Device.objects.select_related(
    'device_type', 'location', 'role'
).all()

PHASE 3: GENERATE CSV
CSV Generation 
import csv
from io import StringIO

def run(self, location=None):
    queryset = Device.objects.select_related(
        'device_type__manufacturer', 'location', 'role'
    )
    if location:
        queryset = queryset.filter(location=location)
    
    # Create CSV in memory
    output = StringIO()
    writer = csv.writer(output)
    
    # Write header
    writer.writerow(['hostname', 'device_type', 'role', 'location'])
    
    # Write data rows
    for device in queryset:
        writer.writerow([
            device.name,
            device.device_type.model,
            device.role.name if device.role else '',
            device.location.name if device.location else '',
        ])
    
    # Log output preview
    csv_content = output.getvalue()
    self.logger.info(f"Exported {queryset.count()} devices")
    for line in csv_content.split('\n')[:5]:
        self.logger.info(line)
    
    return f"Exported {queryset.count()} devices"
