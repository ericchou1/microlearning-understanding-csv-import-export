THE EXPORT JOB
jobs/device_export.py 
import csv
from io import StringIO
from nautobot.apps.jobs import Job, register_jobs, ObjectVar
from nautobot.dcim.models import Device, Location

class DeviceExport(Job):
    class Meta:
        name = "Device CSV Export"
    
    # Optional filter dropdown
    location = ObjectVar(model=Location, required=False)
    
    def run(self, location=None):
        queryset = Device.objects.select_related(
            'device_type', 'location', 'role'
        )
        if location:
            queryset = queryset.filter(location=location)
        
        output = StringIO()
        writer = csv.writer(output)
        writer.writerow(['hostname', 'device_type', 'role', 'location'])
        
        for device in queryset:
            writer.writerow([
                device.name,
                device.device_type.model,
                device.role.name if device.role else '',
                device.location.name if device.location else '',
            ])
        
        self.logger.info(f"Exported {queryset.count()} devices")
        return output.getvalue()

register_jobs(DeviceExport)
