Parsing Logic 
import csv
from io import StringIO

def run(self, csv_file):
    # Decode binary to string
    content = csv_file.read().decode('utf-8')
    
    # Parse with DictReader
    reader = csv.DictReader(StringIO(content))
    
    for row in reader:
        hostname = row['hostname'].strip()
        self.logger.info(f"Found: {hostname}")

from nautobot.dcim.models import Device, DeviceType, Location
from nautobot.extras.models import Role, Status

# Inside run():
active_status = Status.objects.get(name="Active")

for row in reader:
    try:
        device_type = DeviceType.objects.get(model=row['device_type'])
        role = Role.objects.get(name=row['role'])
        location = Location.objects.get(name=row['location'])
        
        device = Device(
            name=row['hostname'],
            device_type=device_type,
            role=role,
            location=location,
            status=active_status,
        )
        device.validated_save()
        self.logger.info(f"‚úì Created: {row['hostname']}")
    except Exception as e:
        self.logger.error(f"‚úó Error: {e}")
