THE FILEVAR PROTOCOL

To import CSV data into Nautobot, you need FileVar‚Äîa special variable that creates a file upload interface in the Job UI.

jobs/device_import.py 
import csv
from io import StringIO
from nautobot.apps.jobs import Job, register_jobs, FileVar
from nautobot.dcim.models import Device, DeviceType, Location
from nautobot.extras.models import Role, Status

class DeviceCSVImport(Job):
    class Meta:
        name = "Device CSV Import"
    
    csv_file = FileVar(description="CSV file to import")
    
    def run(self, csv_file):
        # Decode binary to string
        content = csv_file.read().decode('utf-8')
        reader = csv.DictReader(StringIO(content))
        
        active_status = Status.objects.get(name="Active")
        
        for row in reader:
            device_type = DeviceType.objects.get(model=row['device_type'])
            role = Role.objects.get(name=row['role'])
            location = Location.objects.get(name=row['location'])
            
            device = Device(
                name=row['hostname'],
                device_type=device_type,
                role=role,
                location=location,
                status=active_status,
            )
            device.validated_save()
            self.logger.info(f"‚úì {row['hostname']}")

register_jobs(DeviceCSVImport)
