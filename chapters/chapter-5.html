Step 1: Basic Job Structure

First, Alex creates the basic job structure with a FileVar for CSV upload:

jobs/csv_import.py 
from nautobot.apps.jobs import Job, register_jobs, FileVar


class CSVDeviceImport(Job):
    """Import network devices from a CSV file into Nautobot."""
    
    class Meta:
        name = "CSV Device Import"
        description = "Upload a CSV file to import devices into Nautobot"
    
    csv_file = FileVar(
        description="CSV file with columns: hostname, device_type, role, location"
    )
    
    def run(self, csv_file):
        """Process the uploaded CSV file and create devices."""
        
        # Read file contents
        file_contents = csv_file.read().decode('utf-8')
        self.logger.info(f"File received: {len(file_contents)} bytes")
        
        # For now, just log that it worked
        self.logger.info("File uploaded successfully!")
        
        return "Upload test complete!"


register_jobs(CSVDeviceImport)

üìù Key Components:

  • Job class - Inherits from nautobot.apps.jobs.Job
  • Meta class - Defines job name and description shown in UI
  • FileVar - Creates a file upload field
  • run() - The main method that executes when job runs
  • register_jobs() - Makes the job available in Nautobot

Step 2: Parse the CSV Data

Now Alex adds CSV parsing logic using the skills learned in earlier chapters:

jobs/csv_import.py (updated) 
import csv
from io import StringIO

from nautobot.apps.jobs import Job, register_jobs, FileVar


class CSVDeviceImport(Job):
    """Import network devices from a CSV file into Nautobot."""
    
    class Meta:
        name = "CSV Device Import"
        description = "Upload a CSV file to import devices into Nautobot"
    
    csv_file = FileVar(
        description="CSV file with columns: hostname, device_type, role, location"
    )
    
    def run(self, csv_file):
        """Process the uploaded CSV file and create devices."""
        
        # Read and decode file contents
        file_contents = csv_file.read().decode('utf-8')
        self.logger.info(f"Processing CSV file...")
        
        # Parse CSV using StringIO (treats string as file)
        csv_reader = csv.DictReader(StringIO(file_contents))
        
        # Process each row
        devices_to_create = []
        for row_num, row in enumerate(csv_reader, start=2):
            self.logger.info(f"Row {row_num}: {row}")
            
            # Extract and validate fields
            hostname = row.get('hostname', '').strip()
            device_type = row.get('device_type', '').strip()
            role = row.get('role', '').strip()
            location = row.get('location', '').strip()
            
            if not all([hostname, device_type, role, location]):
                self.logger.warning(f"Row {row_num}: Missing required fields, skipping")
                continue
            
            devices_to_create.append({
                'hostname': hostname,
                'device_type': device_type,
                'role': role,
                'location': location
            })
        
        self.logger.info(f"Found {len(devices_to_create)} valid devices to import")
        
        return f"Parsed {len(devices_to_create)} devices"


register_jobs(CSVDeviceImport)
üí° Notice the StringIO - it lets us use csv.DictReader on a string as if it were a file. Same CSV skills, just a different input source!

Step 3: Create Devices in Nautobot

The final step - actually creating Device objects in Nautobot's database:

jobs/csv_import.py (complete) 
import csv
from io import StringIO

from nautobot.apps.jobs import Job, register_jobs, FileVar
from nautobot.dcim.models import Device, DeviceType, Location
from nautobot.extras.models import Role, Status


class CSVDeviceImport(Job):
    """Import network devices from a CSV file into Nautobot."""
    
    class Meta:
        name = "CSV Device Import"
        description = "Upload a CSV file to import devices into Nautobot"
    
    csv_file = FileVar(
        description="CSV file with columns: hostname, device_type, role, location"
    )
    
    def run(self, csv_file):
        """Process the uploaded CSV file and create devices."""
        
        # Read and decode file contents
        file_contents = csv_file.read().decode('utf-8')
        
        # Parse CSV
        csv_reader = csv.DictReader(StringIO(file_contents))
        
        # Get the "Active" status once
        active_status = Status.objects.get(name="Active")
        
        # Track results
        created = 0
        skipped = 0
        errors = []
        
        for row_num, row in enumerate(csv_reader, start=2):
            hostname = row.get('hostname', '').strip()
            device_type_name = row.get('device_type', '').strip()
            role_name = row.get('role', '').strip()
            location_name = row.get('location', '').strip()
            
            # Skip rows with missing data
            if not all([hostname, device_type_name, role_name, location_name]):
                self.logger.warning(f"Row {row_num}: Missing fields, skipping")
                skipped += 1
                continue
            
            try:
                # Look up related objects
                device_type = DeviceType.objects.get(model=device_type_name)
                role = Role.objects.get(name=role_name)
                location = Location.objects.get(name=location_name)
                
                # Create the device
                device = Device(
                    name=hostname,
                    device_type=device_type,
                    role=role,
                    location=location,
                    status=active_status,
                )
                device.validated_save()
                
                self.logger.info(f"‚úÖ Created device: {hostname}")
                created += 1
                
            except DeviceType.DoesNotExist:
                msg = f"Row {row_num}: DeviceType '{device_type_name}' not found"
                self.logger.error(msg)
                errors.append(msg)
            except Role.DoesNotExist:
                msg = f"Row {row_num}: Role '{role_name}' not found"
                self.logger.error(msg)
                errors.append(msg)
            except Location.DoesNotExist:
                msg = f"Row {row_num}: Location '{location_name}' not found"
                self.logger.error(msg)
                errors.append(msg)
            except Exception as e:
                msg = f"Row {row_num}: Error creating {hostname}: {str(e)}"
                self.logger.error(msg)
                errors.append(msg)
        
        # Summary
        self.logger.info("=" * 50)
        self.logger.info(f"Import Complete!")
        self.logger.info(f"  Created: {created} devices")
        self.logger.info(f"  Skipped: {skipped} rows")
        self.logger.info(f"  Errors: {len(errors)}")
        
        return f"Created {created} devices, {skipped} skipped, {len(errors)} errors"


register_jobs(CSVDeviceImport)
