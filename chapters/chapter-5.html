jobs/prefix_import.py 
import csv
from io import StringIO

from nautobot.apps.jobs import Job, register_jobs, FileVar
from nautobot.ipam.models import Prefix, Namespace
from nautobot.extras.models import Status


class PrefixCSVImport(Job):
    """Import prefixes from CSV."""
    
    class Meta:
        name = "Prefix CSV Import"
        description = "Import IP prefixes from CSV"
    
    csv_file = FileVar(description="CSV with: prefix, namespace, status, description")
    
    def run(self, csv_file):
        content = csv_file.read().decode('utf-8')
        reader = csv.DictReader(StringIO(content))
        
        active_status = Status.objects.get(name="Active")
        created = 0
        skipped = 0
        
        for row in reader:
            prefix_str = row.get('prefix', '').strip()
            namespace_name = row.get('namespace', 'Global').strip()
            description = row.get('description', '').strip()
            
            if not prefix_str:
                continue
            
            # Get or create namespace
            namespace, _ = Namespace.objects.get_or_create(
                name=namespace_name
            )
            
            # Get or create prefix
            prefix, was_created = Prefix.objects.get_or_create(
                prefix=prefix_str,
                namespace=namespace,
                defaults={
                    "status": active_status,
                    "description": description,
                }
            )
            
            if was_created:
                self.logger.info(f"‚úÖ Created prefix: {prefix_str}")
                created += 1
            else:
                skipped += 1
        
        return f"Created {created} prefixes, skipped {skipped} existing"


register_jobs(PrefixCSVImport)

jobs/interface_import.py 
import csv
from io import StringIO

from nautobot.apps.jobs import Job, register_jobs, FileVar
from nautobot.dcim.models import Device, Interface
from nautobot.extras.models import Status


class InterfaceCSVImport(Job):
    """Import interfaces from CSV."""
    
    class Meta:
        name = "Interface CSV Import"
        description = "Import device interfaces from CSV"
    
    csv_file = FileVar(description="CSV with: device, name, type, enabled, description")
    
    def run(self, csv_file):
        content = csv_file.read().decode('utf-8')
        reader = csv.DictReader(StringIO(content))
        
        active_status = Status.objects.get(name="Active")
        created = 0
        errors = []
        
        for row_num, row in enumerate(reader, start=2):
            device_name = row.get('device', '').strip()
            interface_name = row.get('name', '').strip()
            interface_type = row.get('type', '1000base-t').strip()
            enabled = row.get('enabled', 'true').lower() == 'true'
            description = row.get('description', '').strip()
            
            if not device_name or not interface_name:
                continue
            
            # Look up the device
            try:
                device = Device.objects.get(name=device_name)
            except Device.DoesNotExist:
                errors.append(f"Row {row_num}: Device '{device_name}' not found")
                continue
            
            # Get or create interface
            interface, was_created = Interface.objects.get_or_create(
                device=device,
                name=interface_name,
                defaults={
                    "type": interface_type,
                    "enabled": enabled,
                    "description": description,
                    "status": active_status,
                }
            )
            
            if was_created:
                created += 1
        
        self.logger.info(f"Created {created} interfaces")
        if errors:
            self.logger.warning(f"Errors: {len(errors)}")
            for e in errors[:5]:
                self.logger.warning(e)
        
        return f"Created {created} interfaces, {len(errors)} errors"


register_jobs(InterfaceCSVImport)
üí° Note: Interface type values must match Nautobot's interface type choices (e.g., '1000base-t', '10gbase-x-sfpp', 'virtual', etc.)

jobs/ipaddress_import.py 
import csv
from io import StringIO

from nautobot.apps.jobs import Job, register_jobs, FileVar
from nautobot.dcim.models import Device, Interface
from nautobot.ipam.models import IPAddress, Namespace
from nautobot.extras.models import Status


class IPAddressCSVImport(Job):
    """Import IP addresses and assign to interfaces."""
    
    class Meta:
        name = "IP Address CSV Import"
        description = "Import IPs and assign to device interfaces"
    
    csv_file = FileVar(description="CSV with: ip_address, device, interface, description")
    
    def run(self, csv_file):
        content = csv_file.read().decode('utf-8')
        reader = csv.DictReader(StringIO(content))
        
        active_status = Status.objects.get(name="Active")
        default_namespace = Namespace.objects.get(name="Global")
        
        created = 0
        assigned = 0
        errors = []
        
        for row_num, row in enumerate(reader, start=2):
            ip_str = row.get('ip_address', '').strip()
            device_name = row.get('device', '').strip()
            interface_name = row.get('interface', '').strip()
            description = row.get('description', '').strip()
            
            if not ip_str:
                continue
            
            # Get or create IP address
            ip_address, was_created = IPAddress.objects.get_or_create(
                address=ip_str,
                namespace=default_namespace,
                defaults={
                    "status": active_status,
                    "description": description,
                }
            )
            
            if was_created:
                created += 1
                self.logger.info(f"‚úÖ Created IP: {ip_str}")
            
            # Assign to interface if specified
            if device_name and interface_name:
                try:
                    device = Device.objects.get(name=device_name)
                    interface = Interface.objects.get(
                        device=device,
                        name=interface_name
                    )
                    
                    # Add IP to interface (M2M relationship)
                    if ip_address not in interface.ip_addresses.all():
                        interface.ip_addresses.add(ip_address)
                        assigned += 1
                        self.logger.info(f"üîó Assigned {ip_str} to {device_name}:{interface_name}")
                        
                except Device.DoesNotExist:
                    errors.append(f"Row {row_num}: Device '{device_name}' not found")
                except Interface.DoesNotExist:
                    errors.append(f"Row {row_num}: Interface '{interface_name}' not found on {device_name}")
        
        self.logger.info("=" * 50)
        self.logger.info(f"Created: {created} IPs")
        self.logger.info(f"Assigned: {assigned} IPs to interfaces")
        self.logger.info(f"Errors: {len(errors)}")
        
        return f"Created {created} IPs, assigned {assigned}, {len(errors)} errors"


register_jobs(IPAddressCSVImport)

üîó Many-to-Many Relationships

IP addresses have a M2M relationship with interfaces (an IP can be on multiple interfaces, an interface can have multiple IPs). Use .add() to create the relationship:

interface.ip_addresses.add(ip_address)

jobs/full_network_export.py 
import csv
from io import StringIO

from nautobot.apps.jobs import Job, register_jobs, ObjectVar
from nautobot.dcim.models import Device, Location


class FullNetworkExport(Job):
    """Export complete network inventory with IPs."""
    
    class Meta:
        name = "Full Network Export"
        description = "Comprehensive CSV with devices, interfaces, and IPs"
    
    location = ObjectVar(
        model=Location,
        required=False,
        description="Filter by location (optional)"
    )
    
    def run(self, location=None):
        # Build device query with all related data
        queryset = Device.objects.select_related(
            'device_type__manufacturer',
            'location',
            'role',
            'status'
        ).prefetch_related(
            'interfaces',
            'interfaces__ip_addresses',
        ).order_by('name')
        
        if location:
            queryset = queryset.filter(location=location)
        
        output = StringIO()
        writer = csv.writer(output)
        
        # Comprehensive header
        writer.writerow([
            'device',
            'device_type',
            'manufacturer',
            'location',
            'role',
            'device_status',
            'interface',
            'interface_type',
            'interface_enabled',
            'ip_address',
            'ip_status'
        ])
        
        row_count = 0
        
        for device in queryset:
            interfaces = device.interfaces.all()
            
            if not interfaces:
                # Device with no interfaces
                writer.writerow([
                    device.name,
                    device.device_type.model,
                    device.device_type.manufacturer.name,
                    device.location.name if device.location else '',
                    device.role.name if device.role else '',
                    device.status.name if device.status else '',
                    '', '', '', '', ''
                ])
                row_count += 1
            else:
                for interface in interfaces:
                    ip_addresses = interface.ip_addresses.all()
                    
                    if not ip_addresses:
                        # Interface with no IPs
                        writer.writerow([
                            device.name,
                            device.device_type.model,
                            device.device_type.manufacturer.name,
                            device.location.name if device.location else '',
                            device.role.name if device.role else '',
                            device.status.name if device.status else '',
                            interface.name,
                            interface.type,
                            interface.enabled,
                            '', ''
                        ])
                        row_count += 1
                    else:
                        for ip in ip_addresses:
                            writer.writerow([
                                device.name,
                                device.device_type.model,
                                device.device_type.manufacturer.name,
                                device.location.name if device.location else '',
                                device.role.name if device.role else '',
                                device.status.name if device.status else '',
                                interface.name,
                                interface.type,
                                interface.enabled,
                                str(ip.address),
                                ip.status.name if ip.status else ''
                            ])
                            row_count += 1
        
        csv_content = output.getvalue()
        
        self.logger.info(f"Generated report with {row_count} rows")
        self.logger.info(f"Devices: {queryset.count()}")
        self.logger.info("=" * 50)
        self.logger.info("Preview:")
        for line in csv_content.split('\n')[:10]:
            self.logger.info(line)
        
        return f"Exported {row_count} rows from {queryset.count()} devices"


register_jobs(FullNetworkExport)
